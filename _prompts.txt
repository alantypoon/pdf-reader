import { openAsBlob } from 'node:fs';

const endpointUrl = "https://aigateway.aied.hku.hk/api/generate?debug_log=1";

async function runEttExtraction() {
  const formData = new FormData();
  formData.append('provider', 'ett');
  formData.append('apiKey', "<secretly hidden>");
  formData.append('model', "vllm|OpenGVLab/InternVL3_5-38B");
  formData.append('links', JSON.stringify(['https://example.com/article-one', 'https://example.com/article-two']));
  formData.append('wordCount', '1000');
  formData.append('files', await openAsBlob('./brief.pdf'), 'brief.pdf');
  formData.append('files', await openAsBlob('./screenshot.png'), 'screenshot.png');

  const response = await fetch(endpointUrl, {
    method: 'POST',
    headers: { Accept: "text/event-stream" },
    body: formData,
  });

  console.log("HTTP", response.status);
  console.log(await response.text());
}
runEttExtraction().catch(console.error);
